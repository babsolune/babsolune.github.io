Get started

// Template css files
<link href="https://ghcdn.rawgit.org/babsolune/babsolune.github.io/master/FWKBoost/src/base/@import.css" rel="stylesheet" type="text/css" />
 
// jQuery
<script src="https://ghcdn.rawgit.org/babsolune/babsolune.github.io/master/FWKBoost/src/__default__/plugins/jquery.js"></script>
 
// FWKBoost js plugins
<script src="https://ghcdn.rawgit.org/babsolune/babsolune.github.io/master/FWKBoost/src/__default__/plugins/fwkboost.min.js"></script>
 
// js initialization file
<script src="https://ghcdn.rawgit.org/babsolune/babsolune.github.io/master/FWKBoost/src/__default__/plugins/@init.min.js"></script>
