Get started

// Template css files
<link href="templates/base/base.min.css" rel="stylesheet" type="text/css" />
// jQuery
<script src="templates/dist/jquery.js"></script>
// FWKBoost js plugins
<script src="templates/default/fwkboost.min.js"></script>
// Template initialization js file
<script src="templates/base/base.init.js"></script>
